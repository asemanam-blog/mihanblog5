    صفات پارسایان در خطبه متقین امام علی(ع)
    ارسال در تاریخ پنجشنبه 11 خرداد 1391 توسط حمیدرضا عرب

    بسم الله الرحمن الرحیم

    رُوِیَ أنَّ صَاحِباً لِأمِیرِ الْمُؤْمِنِینَ علیه السلام یُقالُ لَهُ: هَمَّامٌ كانَ رَجُلاً عابداً،فَقالَ لَهُ یا أمِیرَالْمُؤْمِنِینَ صِفْ لِیَ الْمُتَّقِینَ حَتَّى كَأنِّی أنْظُرُ إِلَیْهِمْ،فَتَثَاقَلَ علیه السلام عَنْ جَوَابهِ،ثُمَّ قَالَ یا هَمَّامُ إتَّقِ اللَّهَ وَأحْسِنْ (فَإِنَّ اللّهَ مَعَ الَّذِینَ اتَّقَوْا وَ الَّذِینَ هُمْ مُحْسِنُونَ

    راوی می گوید كه امیرالمؤ منین علیه السلام مصاحبى داشت به نام همّام كه مردى عبادت پیشه بود روزى گفتش كه اى امیرالمؤ منین ، پرهیزگاران را برایم توصیف كن، آنسان كه گویى در آنها مى نگرم امیرالمؤ منین علیه السلام در پاسخش درنگ كرد، سپس گفت: اى همّام از خدا بترس و نیكوكار باش كه خدا با كسانى است كه پرهیزگارى كنند و نیكوكارند

     

    فَلَمْ یَقْنَعْ هَمَّامٌ بِذِلِكَ الْقَوْلِ حَتَّى عَزَمَ عَلَیْهِ،فَحَمِدَ اللَّهَ وَ أثْنَى عَلَیْهِ وَ صَلَّى عَلَى النَّبِیِّ صلی الله علیه و آله ثُمَّ قَالَ
    همّام بدین سخن قانع نشد و امیرالمؤمنین علیه السلام را سوگند داد
    امیرالمؤمنین علیه السلام حمد و ثناى خداى به جاى آورد و بر محمد (صلی الله و علیه و آله) و خاندانش درود فرستاد سپس فرمود

    أمَّا بَعْدُ، فَإِنَّ اللَّهَ سُبْحانَهُ وَ تَعالَى خَلَقَ الْخَلْقَ حِینَ خَلَقَهُمْ غَنِیّا عَنْ طاعَتِهِمْ، آمِنا مِنْ مَعْصِیَتِهِمْ، لِأنَّهُ لا تَضُرُّهُ مَعْصِیَةُ مَنْ عَصاهُ، وَ لا تَنْفَعُهُ طاعَةُ مَنْ أَطاعَهُ، فَقَسَمَ بَیْنَهُمْ مَعایِشَهُمْ، وَ وَضَعَهُمْ مِنَ الدُّنْیا مَواضِعَهُمْ
    اما بعد، خداوند، سبحانه و تعالى ، موجودات را بیافرید، و چون بیافرید از فرمانبرداریشان بى نیاز بود و از نافرمانیشان در امان، زیرا نه نافرمانى نافرمایان او را زیانى رساند و نه فرمانبردارى فرمانبرداران سودى آنگاه روزیهایشان را میانشان تقسیم كرد و جاى هر یك را در این جهان معین ساخت

    فَالْمُتَّقُونَ فِیها هُمْ أهْلُ الْفَضائِلِ، مَنْطِقُهُمُ الصَّوابُ، وَ مَلْبَسُهُمُ الاِقْتِصادُ، وَ مَشْیُهُمُ التَّواضُعُ

    پس پرهیزگاران را در این جهان فضیلتهاست، گفتارشان به صواب مقرون و راه و رسمشان بر اعتدال و رفتارشان با فروتنى آمیخته است

    غَضُّوا أبْصارَهُمْ عَمَّا حَرَّمَ اللَّهُ عَلَیْهِمْ، وَ وَقَفُوا أسْماعَهُمْ عَلَى الْعِلْمِ النَّافِعِ لَهُمْ، نُزِّلَتْ أنْفُسُهُمْ مِنْهُمْ فِی الْبَلاءِ كَالَّتِی نُزِّلَتْ فِی الرَّخاءِ
    از هر چه خداوند بر آنها حرام كرده است ، چشم مى پوشند و گوش بر دانستن چیزى نهاده اند كه آنان را سودى رساند. آنچنان به بلا خو گرفته اند كه گویى در آسودگى هستند

    وَ لَوْ لا الْأجَلُ الَّذِی كَتَبَ اللَّهُ لَهُمْ لَمْ تَسْتَقِرَّ أرْواحُهُمْ فِی أجْسادِهِمْ طَرْفَةَ عَیْنٍ شَوْ قا إِلَى الثَّوابِ، وَ خَوْ فا مِنَ الْعِقابِ
    اگر مدت عمرى نبود كه خداوند برایشان مقرر داشته ، به سبب شوقى كه به پاداش نیك و بیمى كه از عذاب روز بازپسین دارند، چشم بر هم زد نى جانهایشان در بدنهایشان قرار نمى گرفت

    عَظُمَ الْخالِقُ فِی أنْفُسِهِمْ فَصَغُرَ ما دُونَهُ فِی أعْیُنِهِم

    تنها آفریدگار در نظرشان بزرگ است و جز او هر چه هست در دیدگانشان خرد مى نماید

    فَهُمْ وَ الْجَنَّةُ كَمَنْ قَدْ رَآها فَهُمْ فِیها مُنَعَّمُونَ، وَ هُمْ وَ النَّارُ كَمَنْ قَدْ رَآها فَهُمْ فِیها مُعَذَّبُونَ با بهشت چنان اند كه گویى مى بینندش و غرق نعمتهایش هستند. و با دوزخ چنانند كه گویى مى بینندش و به عذاب آن گرفتارند

    قُلُوبُهُمْ مَحْزُونَةٌ، وَ شُرُورُهُمْ مَأْمُونَةٌ، وَ أجْسادُهُمْ نَحِیفَةٌ، وَ حاجاتُهُمْ خَفِیفَةٌ، وَ أ نْفُسُهُمْ عَفِیفَةٌ

    دلهایشان اندوهگین است و مردم از آسیب شان در امان اند و بدنهاشان لاغر است و نیازهاشان اندك است و جانشان به زیور عفت آراسته است

    صَبَرُوا أیَّاما قَصِیرَةً أعْقَبَتْهُمْ راحَةً طَوِیلَةً، تِجارَةٌ مُرْبِحةٌ یَسَّرَها لَهُمْ رَبُّهُمْ، أرادَتْهُمُ الدُّنْیا فَلَمْ یُرِیدُوها، وَ أسَرَتْهُمْ فَفَدَوْا أنْفُسَهُمْ مِنْهَا

    چند روز كوتاه دنیا را شكیبایی كرده تا آسایشى ابدى قیامت را به دست آورند، این تجارت ، كه پروردگارشان نیز برایشان فراهم ساخته، سود بسیار دهد. دنیا در طلب آنهاست و آنها از دنیاگریزان اند، به اسارتشان مى گیرد ولى جانهاى خویش به فدیه دهند تا از اسارت برهند

    أمَّا اللَّیْلَ، فَصافُّونَ أقْدامَهُمْ تالِینَ لِأجْزاءِ الْقُرْآنِ، یُرَتِّلُو نَها تَرْتِیلاً، یُحَزِّنُونَ بهِ أنْفُسَهُمْ، وَ یَسْتَثِیرُونَ بهِ دَواءَ دائِهِمْ

     اما شبها، همچنان برپاى ایستاده اند تا جزء جزء كتاب خدا را بخوانند، و آرام و با تأنّى و تدبّر مى خوانند. به هنگام خواندنش خود را اندوهگین مى سازند و داروى درد خویش از آن مى جویند

    وَ إِذا مَرُّوا بِآیَةٍ فِیها تَخْوِیفٌ أصْغَوْا إِلَیْها مَسامِعَ قُلُوبِهِمْ، وَ ظَنُّوا أنَّ زَفِیرَ جَهَنَّمَ وَ شَهِیقَها فِی أُصُولِ آذانِهِمْ

    و چون به آیتى رسند كه در آن وعید عذاب باشد گوش دل بدان مى سپارند و پندارند كه اكنون بانگ جوش و خروش جهنم در گوششان پیچیده است

    فَهُمْ حانُونَ عَلَى أوْساطِهِمْ، مُفْتَرِشُونَ لِجَباهِهِمْ وَ أكُفِّهِمْ وَ رُكَبِهِمْ وَ أطْرافِ أقْدَامِهِمْ، یَطْلُبُونَ إِلَى اللَّهِ تَعالى فِی فَكاكِ رِقابِهِمْ

    در برابر پروردگارشان ركوع می كنند و پیشانى و كف دست و زانو و نوك پاى بر زمین نهاده اند و از خداوند تعالى آزادی خود را از آتش جهنّم مى طلبند

    وَ أمَّا النَّهارَ، فَحُلَماءُ عُلَماءُ أبْرارٌ أتْقِیَاءُ، قَدْ بَراهُمُ الْخَوْفُ بَرْیَ الْقِداحِ، یَنْظُرُ إِلَیْهِمُ النَّاظِرُ فَیَحْسَبُهُمْ مَرْضَى وَ ما بِالْقَوْمِ مِنْ مَرَض

    اما در روزها، دانشمندانی بردبار، نیكوكارانی، پرهیزكار هستند كه ترس از خدا آنان را چونان تیر تراشیده لاغر كرده است، چون بیننده اى در آنان نگرد، پندارد كه بیمارند و حال آنكه ، بیمار نیستند

    وَیَقُولُ لَقَدْ خُولِطُوا وَ لَقَدْ خالَطَهُمْ أمْرٌ عَظِیمٌ، لا یَرْضَوْنَ مِنْ أعْمالِهِمُ الْقَلِیلَ،وَلا یَسْتَكْثِرُونَ الْكَثِیرَ

    و مردم به اشتباه گویند بى شك در عقلشان خللى است، آرى، كارى بزرگ آنها را به خود مشغول داشته از اعمال خویش چون اندك باشد، ناخشنودند و چون بسیار باشد در نظرشان اندك نماید

    فَهُمْ لِأنْفُسِهِمْ مُتَّهِمُونَ، وَ مِنْ أعْمالِهِمْ مُشْفِقُونَ، إِذا زُكِّیَ أحَدٌ مِنْهُمْ خافَ مِمَّا یُقالُ لَهُ، فَیَقُولُ: أنَا أعْلَمُ بِنَفْسِی مِنْ غَیْرِی ، وَ رَبِّیأعْلَمُ بِی مِنِّی بِنَفْسِی ؛ اللَّهُمَّ لا تُؤَاخِذْنِی بِما یَقُولُونَ، وَ اجْعَلْنِی أفْضَلَ مِمَّا یَظُنُّونَ، وَ اغْفِرْ لِی ما لا یَعْلَمُونَ

    كه اینان پیوسته خود را متهم مى دارند و از آنچه مى كنند بیمناك اند.چون یكی از آنان را به پاكى بستایند، از آنچه درباره اش ‍ مى گویند بیمناك مى شود و مى گوید كه من خود به خویشتن آگاهترم و پروردگار من به من از من آگاهتر است، اى پروردگار من ، مرا به آنچه مى گویند مؤ اخذه مكن و مرا بهتر از آنچه مى پندارند بگردان و گناهان مرا كه از آن بى خبرند، بیامرز

    فَمِنْ عَلامَةِ أحَدِهِمْ: أنَّكَ تَرَى لَهُ قُوَّةً فِی دِین، وَ حَزْماً فِی لِین وَ إِیمَاناً فِی یَقِینٍ، وَ حِرْصاً فِی عِلْمٍ

    از نشانه هاى یكی از آنها این است كه مى بینى: در دینداری نیرومند است و در عین دوراندیشى نرمخوى و ایمانش ‍ پر از یقین است و در كسب علم آزمند

    وَ عِلْماً فِی حِلْمٍ، وَ قَصْداً فِی غِنًى ، وَ خُشُوعاً فِی عِبادَةٍ، وَ تَجَمُّلاً فِی فاقَةٍ، وَ صَبْراً فِی شِدَّةٍ، وَ طَلَباً فِی حَلالٍ وَ نَشاطاً فِی هُدىً وَ تَحَرُّجاً عَنْ طَمَع

    و علمش آمیخته به حلم و توانگریش ‍ همراه با میانه روى است و عبادتش پیوسته با خشوع، در بینوایى محتشم است و در سختى ، بردبار، در طلب حلال است و در جستجوى هدایت شادمان و از آزمندى به دور است

    یَعْمَلُ الْأعْمالَ الصَّالِحَةَ وَ هُوَ عَلَى وَجَلٍ، یُمْسِی وَ هَمُّهُ الشُّكْرُ، وَ یُصْبِحُ وَ هَمُّهُ الذِّكْرُ

    در حالی كه به كارهاى شایسته مى پردازد، دلش بیمناك است . سپاس گویان روز را به شب مى آورد و ذكرگویان شب را به روز مى رساند

    یَبِیتُ حَذِراً، وَ یُصْبِحُ فَرِحاً: حَذِراً لَمَّا حُذِّرَ مِنَ الْغَفْلَةِ، وَ فَرِحاً بِما أصابَ مِنَ الْفَضْلِ وَ الرَّحْمَةِ

    شب را در عین هراس مى گذراند و شادمانه دیده به دیدار صبح مى گشاید، هراسش از غفلتى است كه مبادا گریبانگیرش شود و شادمانیش از فضل و رحمتى است كه نصیبش گشته

    إِن اسْتَصْعَبَتْ عَلَیْهِ نَفْسُهُ فِیما تَكْرَهُ لَمْ یُعْطِها سُؤْلَها فِیما تُحِبُّ، قُرَّةُ عَیْنِهِ فِیما لا یَزُولُ، وَ زَهادَتُهُ فِیما لا یَبْقَى یَمْزُجُ الْحِلْمَ بِالْعِلْمِ، وَ الْقَوْلَ بِالْعَمَلِ

    اگر نفسش در طلب چیزى ناخوشایند سركشى كند، پاى مى فشرد تا خواهشش را برنیاورد. شادمانى دلش چیزى است كه پایدار است و پرهیزش ، از چیزى كه نمى پاید. دانش را به بردبارى و گفتار را با كردار آمیخته است

    تَراهُ قَرِیبا أمَلُهُ، قَلِیلاً زَلَلُهُ، خاشِعا قَلْبُهُ، قانِعَةً نَفْسُهُ، مَنْزُورا أكْلُهُ، سَهْلاً أمْرُهُ، حَرِیزا دِینُهُ، مَیِّتَةً شَهْوَتُهُ مَكْظُوما غَیْظُهُ، الْخَیْرُ مِنْهُ مَأمُولٌ، وَ الشَّرُّ مِنْهُ مَأمُونٌ

    او را بینى كه آرزویش كوتاه است و خطایش اندك، دلش خاشع و نفسش قانع است، خوردنش اندك و كارهایش آسان است و دینش محفوظ و امیالش در حرام مرده و خشمش فرو خورده، به خیرش امید و از شرّش ایمنى است

    إِنْ كانَ فِی الْغافِلِینَ كُتِبَ فِی الذَّاكِرِینَ، وَ إِنْ كانَ فِی الذَّاكِرِینَ لَمْ یُكْتَبْ مِنَ الْغافِلِینَ

    اگر در جمع غافلان باشد، نامش را در زمره ذاكران نویسند و اگر در میان ذاكران باشد، در شمار غافلانش نیاورند

    یَعْفُو عَمَّنْ ظَلَمَهُ، وَ یُعْطِی مَنْ حَرَمَهُ، وَ یَصِلُ مَنْ قَطَعَهُ، بَعِیدا فُحْشُهُ، لَیِّناً قَوْلُهُ، غَائِباً مُنْكَرُهُ حاضِراً مَعْرُوفُهُ، مُقْبِلاً خَیْرُهُ، مُدْبراً شَرُّهُ، فِی الزَّلازِلِ وَقُورٌ، وَ فِی الْمَكارِهِ صَبُورٌ، وَ فِی الرَّخاءِ شَكُورٌ

    اگر بر او ستمى رود، عفو كند و به آن كس كه محرومش داشته ، بخشش نماید. و با هر كه از او ببرد، پیوند كند. زشتگویى از او دور است، گفتارش نرم است، بدیهای او پنهان و نیكوكارى در او هویدا است همواره خیرش به همه رسیده و آزار او به كسی نمی رسد، در سختی ها آرام و در ناگواری ها شكیبا و خوشی ها سپاسگزار ‍ است

    لا یَحِیفُ عَلَى مَنْ یُبْغِضُ، وَ لا یَأثَمُ فِیمَنْ یُحِبُّ، یَعْتَرِفُ بِالْحَقِّ قَبْلَ أنْ یُشْهَدَ عَلَیْهِ                  

    بر كسى كه دشمن دارد ستم روا ندارد و محبت دیگران به گناهش نكشاند.پیش از آنكه بر زیانش شهادت دهند، او خود به حقیقت اعتراف مى كند

    لا یُضِیعُ مَا اسْتُحْفِظَ، وَ لایَنْسَى ما ذُكِّرَ، وَ لا یُنابِزُ بالْألْقابِ، وَ لا یُضارُّ بالْجارِ، وَ لا یَشْمَتُ بالْمَصائِبِ، وَ لا یَدْخُلُ فِی الْباطِلِ وَ لا یَخْرُجُ مِنَ الْحَقِّ

    و آنچه را به او سپرده اند ضایعش ‍ نمى كند، آنچه را كه به او تذكر دادند از یاد نمى برد، دیگران را با القاب زشت نمى خواند،به همسایه زیان نمى رساند، به هنگام مصیبت شماتت روا نمى دارد، به باطل وارد نمى شود و از محدوده حق خارج نمی شود

    إِنْ صَمَتَ لَمْ یَغُمَّهُ صَمْتُهُ، وَ إِنْ ضَحِكَ لَمْ یَعْلُ صَوْتُهُ، وَ إِنْ بُغِیَ عَلَیْهِ صَبَرَ حَتَّى یَكُونَ اللَّهُ هُوَ الَّذِی یَنْتَقِمُ لَهُ

    اگر خاموش باشد از خاموشى خویش غمگین نمى گردد. اگر بخندد صدای خنده او بلند نمى شود، چون بر او ستمى رود، صبر مى كند تا خدا انتقامش ‍ را بستاند

    نَفْسُهُ مِنْهُ فِی عَناءٍ، وَ النَّاسُ مِنْهُ فِی راحَةٍ، أتْعَبَ نَفْسَهُ لِآخِرَتِهِ، وَ أراحَ النَّاسَ مِنْ نَفْسِهِ

    خود را به رنج مى افكند و مردم از او در راحت اند.براى قیامت ، خود را به مشقت مى اندازد و مردم را به رفاه و آسایش مى رساند

    بُعدُهُ عَمَّنْ تَباعَدَ عَنْهُ زُهْدٌ وَ نَزاهَةٌ، وَ دُنُوُّهُ مِمَّنْ دَنا مِنْهُ لِینٌ وَ رَحْمَةٌ، لَیْسَ تَباعُدُهُ بِكِبْرٍ وَ عَظَمَةٍ، وَ لا دُنُوُّهُ بِمَكْرٍ وَ خَدِ یعَةٍ

    از هر كه دورى گزیند به سبب پارسایى و پاكى است و به هر كه نزدیك شود به سبب نرمخویى و رحمت است نه دورى گزیدنش از روى تكبر و نه نزدیك شدنش از روى مكر و خدعه است

    قالَ :فَصَعِقَ هَمَّامٌ صَعْقَةً كانَتْ نَفْسُهُ فِیها

    راوی گوید كه : همّام از این سخن بیهوش شد و در آن بیهوشى جان داد

    فَقالَ أمِیرُ الْمُؤْمِنِینَ علیه السلام : أما وَ اللَّهِ لَقَدْ كُنْتُ أخافُها عَلَیْهِ ،ثُمَّ قالَ: هَكذا تَصْنَعُ الْمَواعِظُ الْبالِغَةُ بأهْلِها

    امیرالمؤ منین علیه السلام گفت : كه بر جانش بیمناك بودم سپس فرمود: آرى ، اندرزهاى رسا به هر كه اهلش باشد چنین كند

    فَقالَ لَهُ قَائِلٌ :فَما بالُكَ یا أمِیرَ الْمُؤْمِنِینَ؟

    یكى گفت یا امیرالمؤ منین تو خود چگونه اى ؟ یعنی چرا در خودت اثر نمی كند

    فَقالَ علیه السلام : وَیْحَكَ! إِنَّ لِكُلِّ أجَلٍ وَقْتا لا یَعْدُوهُ، وَ سَبَبا لا یَتَجاوَزُهُ، فَمَهْلاً لا تَعُدْ لِمِثْلِها، فَإِنَّما نَفَثَ الشَّیْطانُ عَلَى لِسانِكَ

    امیرالمؤ منین علیه السلام گفت : واى بر تو، مرگ هر كس را زمانى است كه از او در نگذرد و سبب مشخصی دارد كه از آن تجاوز نكند آرام باش و دیگر چنین سخنانی مگو كه شیطان آن را بر زبان تو دمیده است
